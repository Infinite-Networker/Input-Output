🔌 Developer API

Integrate in Minutes

A clean, well-documented REST API with SDKs for Python and JavaScript.

import inputoutput as io

# Initialize the client
client = io.Client(api_key="your_api_key")

# Semantic search with license filtering
results = client.search(
    query="machine learning in healthcare",
    filters={
        "license": ["CC BY", "CC BY-SA"],
        "peer_reviewed": True,
        "date_from": "2023-01-01"
    }
)

# Auto-generate a citation
for paper in results:
    print(paper.cite("APA"))

import { InputOutput } from '@cherry-computer/input-output-sdk';

// Initialize the client
const io = new InputOutput({
  apiKey: process.env.IO_API_KEY
});

// Search and get citations
const results = await io.search.query({
  q: "machine learning in healthcare",
  license: ["CC BY"],
  peerReviewed: true
});

results.forEach(paper => {
  console.log(paper.cite("APA"));
});

# Search for open-access papers
curl -X GET \
  "https://api.input-output.io/v1/search?\
q=machine+learning+healthcare&\
license=CC+BY&\
peer_reviewed=true" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json"

# Response: JSON array of papers with
# full license info and citation data
